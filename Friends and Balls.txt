#include<bits/stdc++.h>
using namespace std;

int main(){
    int n,m;
    cin>>n>>m;
    
    vector<int>arr(n,0);
    
    while(m--){
        int l,r;
        cin>>l>>r;
        
        for(int i=l-1;i<r;i++){
            arr[i]++;
        }
    }
    
    // for(int i=0;i<n;i++){
    //     cout<<arr[i]<<" ";
    // }
    
    int q;
    cin>>q;
    
    while(q--){
        int k;
        cin>>k;
        
        int ans = 0;
        
        for(int i=0;i<arr.size();i++){
            if(arr[i]>=k){
                ans++;
            }
        }
        
        cout<<ans<<endl;
    }
}
